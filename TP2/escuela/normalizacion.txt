create table Jurisdiccion(
	id integer not null primary key autoincrement,
	nombre varchar(64) not null
)

create table Departamento(
	id integer not null primary key autoincrement,
	nombre varchar(64) not null,
	jurisdiccion_id int not null,
	
	foreign key (jurisdiccion_id) references Jurisdiccion(id)
);

create table Sector(
	id integer not null primary key autoincrement,
	nombre varchar(64) not null
);

create table Localidad(
	id integer not null primary key autoincrement,
	nombre varchar(64) not null,
	codigo_postal int not null,
	codigo_localidad int not null,
	departamento_id int not null,

	foreign key (departamento_id) references Departamento(id)
);

create table Domicilio(
	id integer not null primary key autoincrement,
	nombre varchar(64) not null,
	localidad_id int not null,
	
	foreign key (localidad_id) references Localidad(id)
);

create table Codigo_de_area(
	id integer not null primary key autoincrement,
	codigo int not null
);

create table Telefono(
	id integer not null primary key autoincrement,
	codigo_de_area_id int null,
	numero_telefono int not null,

	foreign key (codigo_de_area_id) references codigo_de_area(id)
);

create table Ambito(
	id integer not null primary key autoincrement,
	nombre varchar(64) not null
);

create table Escuela(
	id integer not null primary key autoincrement,
	Nombre varchar(64) not null,
	CUE_Anexo int not null unique,
	mail varchar(64) null,

	codigo_area_id int null,
	domicilio_id int null,
	telefono_id int null,
	sector_id int not null,
	ambito_id int null,
	
	foreign key (codigo_de_area_id) references Codigo_de_area(id),
	foreign key (domicilio_id) references Domicilio(id),
	foreign key (telefono_id) references Telefono(id),
	foreign key (sector_id) references Sector(id),
	foreign key (ambito_id) references Ambito(id)
);
