--Plan
select p.nombre as Nombre, a.nombre as Asignatura, n.numero as Nivel, c.descripcion as Cuatrimestre, ar.nombre as Area
from MateriasPlan mp
	inner join Plan p
	on p.id = mp.plan_id
	inner join Asignatura a
	on a.id = mp.asignatura_id
	inner join Nivel n
	on n.id = mp.nivel_id
	inner join Cuatrimestre c
	on c.id = mp.cuatrimestre_id
	inner join Area ar
	on a.area_id = ar.id;
